 13def mfForecast(
 14    mfmodel: MFESN, 
 15    forecast_data, 
 16    fit: dict,
 17    ref_dates=None,
 18    stack_init=True,
 19    **kwargs
 20) -> dict:
 21    """
 22    Forecast using MFESN model.
 23    mfmodel: MFESN class model to forecast with.
 24    forecast_data: dataset of regressors to use for forecasting.
 25    fit: MFESN fit that contains parameter matrix W.
 26    ref_dates: slicing dates for states collected with forecast_data. 
 27               If not set, the function will try to slice align states 
 28               according to lowest-frequency input data in forecast_data.
 29    stack_init: optionally stack init values when making forecast.
 30    """
 31    # prepare data
 32    Zf, Zf_dates = pd_data_prep(forecast_data)
 33
 34    # handle kwargs
 35    init = None if not 'init' in kwargs.keys() else kwargs['init']
 36
 37    # load init states and dates
 38    init_list = []
 39    init_dates = []
 40    if init is None:
 41        # init = [np.zeros(n) for n in mfmodel.N]
 42        # slice fit states as init using dates
 43        for j, Xj in enumerate(fit['states']):
 44            init_list.append(Xj.iloc[-1,:].to_numpy().squeeze())
 45            cpd = Xj.index[-1]
 46            init_dates.append(cpd)
 47    else:
 48        for j, initj in enumerate(init):
 49            d = Zf_dates[j]
 50            cpd, _ = closest_past_date(Xj.index, d, strict=True)
 51            init_list.append(initj.loc[cpd].to_numpy().squeeze())
 52            init_dates.append(cpd)
 53
 54    # Shape checks
 55    assert fit['W'].shape[0] == mfmodel.M+1, "fit['W'] and MFESN state dimensions are not compatible"
 56    W = fit['W']
 57
 58    # forecast states
 59    mf_forecast_states = mf_dt_states(
 60        input=Zf, 
 61        mfmodel=mfmodel, 
 62        dates=Zf_dates, 
 63        init=init_list, 
 64    )
 65
 66    # find minimum frequency
 67    state_ref_dates = None
 68    if ref_dates is None:
 69        state_ref_dates = pd_min_freq_dates(Zf_dates)
 70    else:
 71        assert isinstance(ref_dates, pd.DatetimeIndex), "Supplied ref_dates is not a pd.DatetimeIndex"
 72        state_ref_dates = ref_dates
 73
 74    # slice and stack model states plus slicing dates
 75    mf_X0_f, mf_X0_f_dates = mf_dt_states_to_matrix(
 76        states=mf_forecast_states, 
 77        ref_dates=state_ref_dates, 
 78        states_join=mfmodel.states_join, 
 79        states_lags=mfmodel.states_lags,
 80    )
 81
 82    # stack init
 83    if stack_init:
 84        mf_X_f = np.vstack([
 85            fit['mf_X'][[-1],:], mf_X0_f,
 86        ])
 87        forecast_dates = pd.DatetimeIndex([]).union([pd_min_freq_dates(fit['mf_X_dates'])[-1]]).union(state_ref_dates)
 88    else:
 89        mf_X_f = mf_X0_f
 90        forecast_dates = state_ref_dates
 91
 92    # Forecast
 93    T = mf_X_f.shape[0]
 94    mf_X_n = np.hstack((np.ones((T, 1)), mf_X_f))
 95    forecast = pd.DataFrame(data=(mf_X_n @ W), index=forecast_dates)
 96
 97    return({
 98        'model': "MFESN",
 99        'method': "mfForecast",
100        'states': mf_forecast_states,
101        'mf_X0_f': mf_X0_f,
102        'mf_X0_f_dates': mf_X0_f_dates,
103        'mf_X_f': mf_X_f,
104        'forecast': forecast,
105        # additional info
106        'init': init,
107    })

109def mfDirectHighFreqForecast(
110    mfmodel: MFESN, 
111    forecast_data, 
112    fit: dict,
113    freqs=None,
114    terminal=True,
115    **kwargs
116) -> dict:
117    """
118    Forecast using MFESN model at high frequency.
119    mfmodel: MFESN class model to forecast with.
120    forecast_data: dataset of regressors to use for forecasting.
121    fit: MFESN fit that contains parameter matrix W.
122    ref_dates: slicing dates for states collected with forecast_data. 
123               If not set, the function will try to slice align states 
124               according to lowest-frequency input data in forecast_data.
125    stack_init: optionally stack init values when making forecast.
126    """
127    # prepare data
128    Zf, Zf_dates = pd_data_prep(forecast_data)
129
130    # handle kwargs
131    init = None if not 'init' in kwargs.keys() else kwargs['init']
132
133    # load init states and dates
134    init_list = []
135    init_dates = []
136    if init is None:
137        # init = [np.zeros(n) for n in mfmodel.N]
138        # slice fit states as init using dates
139        for j, Xj in enumerate(fit['states']):
140            init_list.append(Xj.iloc[-1,:].to_numpy().squeeze())
141            cpd = Xj.index[-1]
142            init_dates.append(cpd)
143    else:
144        for j, initj in enumerate(init):
145            d = Zf_dates[j]
146            cpd, _ = closest_past_date(Xj.index, d, strict=True)
147            init_list.append(initj.loc[cpd].to_numpy().squeeze())
148            init_dates.append(cpd)
149
150    # TODO: currently, due to complexity, only state alignment option 'align'
151    #       is implemented below
152    assert mfmodel.states_join == "align", "mfDirectHighFreqForecast() currently only supports states_join='align'"
153
154    # forecast steps
155    steps_idx = [i for i in range(len(fit['fits'])) if fit['fits'][i]['s'] > 0]
156    steps = [fit['fits'][i]['s'] for i in steps_idx]
157    
158    # Shape checks
159    if fit['method'] == 'mfRidgeFit.fit':
160        assert fit['W'].shape[0] == mfmodel.M+1, "fit['W'] and MFESN state dimensions are not compatible"
161        W = fit['W']
162
163    elif fit['method'] == 'mfRidgeFit.fitDirectHighFreq':
164        # check for all 
165        W_list = []
166        for i in steps_idx:
167            W_list_s = []
168            for j, fitj in enumerate(fit['fits'][i]['freq_fits']):
169                assert fitj['W'].shape[0] == mfmodel.M+1, f"fit['fits'][{j}]['W'] and MFESN state dimensions are not compatible"
170                W_list_s.append(fitj['W'])
171            # NOTE: cycle last element to correspond to
172            #       modulus == 0 c.f. below
173            W_list_s = [W_list_s[-1], ] + W_list_s[:-1]
174            W_list.append(W_list_s)
175        W = None
176
177    # forecast states
178    # NOTE: these are constructed directly here because one
179    #       must handle cases where some data is missing due
180    #       to HF forecasting setup
181    Xfhff = []
182    for j, Zfj in enumerate(Zf):
183        if len(Zfj) == 0:
184            # HF forecasting data for jth component is not available
185            # just yield last collected state when fitted
186            fstates_j = pd.DataFrame(
187                data=init_list[j][None,:], 
188                index=[init_dates[j]],
189            )
190        else:
191            # collect states
192            fstates_j = states(
193                Zfj, 
194                map=mfmodel.models[j].pars.smap, 
195                A=mfmodel.models[j].pars.A, 
196                C=mfmodel.models[j].pars.C, 
197                zeta=mfmodel.models[j].pars.zeta, 
198                rho=mfmodel.models[j].pars.rho, 
199                gamma=mfmodel.models[j].pars.gamma, 
200                leak=mfmodel.models[j].pars.leak, 
201                init=init_list[j],
202            )
203            # add dates to states
204            fstates_dates_j = pd.DatetimeIndex([]).union([init_dates[j]]).union(Zf_dates[j])
205            fstates_j = pd.DataFrame(
206                data=np.vstack([init_list[j], fstates_j]), 
207                index=fstates_dates_j,
208            )
209        # append
210        Xfhff.append(fstates_j)
211
212    # find high-frequency dates
213    if freqs is None:
214        # no suggestion, need to infer from data
215        max_freq_dates = pd_max_freq_dates(Zf_dates)
216    else:
217        for f in freqs:
218            assert type(f) is str, "All frequencies in 'freqs' must be strings"
219        max_freq_j = pd_max_freq(freqs, return_index=True)
220        max_freq_dates = Zf_dates[max_freq_j]
221
222    # high-freq forecast
223    hffor = []
224    if terminal:
225        # only forecast using the terminal state
226        T = 1
227        hff_X = np.full((1, mfmodel.M), np.nan)
228        hff_X_dates = []
229
230        p = 0
231        for j, Xfhffj in enumerate(Xfhff):
232            cpd, _ = closest_past_date(Xfhffj.index, max_freq_dates[-1])
233            hff_X[0,p:(p + mfmodel.N[j])] = np.squeeze(Xfhffj.loc[cpd])
234            # save slice date
235            hff_X_dates.append(cpd)
236            p += mfmodel.N[j]
237
238        hfforecast_dates = [max_freq_dates[-1]]
239
240        for i, s in enumerate(steps):
241            if fit['method'] == 'mfRidgeFit.fitDirectHighFreq':
242                # must choose the right fit to nowcast
243                # NOTE: this is a bit hacky, but we recycle the last index date
244                #       from the first fit as a reference and count the division
245                #       residual to terminal date
246                fit_lf_last_date = fit['fits'][i]['freq_fits'][0]['mf_Y'].index[-1]
247                h = len(max_freq_dates[max_freq_dates >= fit_lf_last_date]) % fit['freqratio'] 
248                # select step W
249                W_s = W_list[i][h]
250
251            mf_X_hff = np.hstack((np.ones((1, 1)), hff_X))
252            hfforecast = pd.DataFrame(data=(mf_X_hff @ W_s), index=hfforecast_dates)
253
254            # output
255            hffor.append({
256                's': s,
257                'forecast': hfforecast
258            })
259
260    else:
261        # nowcast using all collected states
262        T = len(max_freq_dates)
263        hff_X = np.full((T, mfmodel.M), np.nan)
264        hff_X_dates = []
265
266        p = 0
267        for j, Xfnj in enumerate(Xfhff):
268            kt = 0
269            cpdl_j = []
270            for t, hf_date_t in enumerate(max_freq_dates):
271                cpd, kt = closest_past_date(Xfnj.index, hf_date_t, cutoff=kt)
272                hff_X[t,p:(p + mfmodel.N[j])] = np.squeeze(Xfnj.loc[cpd])
273                # save slice date
274                cpdl_j.append(cpd)
275            p += mfmodel.N[j]
276            # save alignment dates
277            hff_X_dates.append(pd.to_datetime(cpdl_j))
278
279        hfforecast_dates = max_freq_dates
280
281        for i, s in enumerate(steps):
282            if fit['method'] == 'mfRidgeFit.fitDirectHighFreq':
283                # must choose the right fit to *initiate* nowcast
284                # NOTE: see above
285                fit_lf_last_date = fit['fits'][0]['freq_fits'][0]['mf_Y'].index[-1]
286                h = len(max_freq_dates[max_freq_dates >= fit_lf_last_date]) % fit['freqratio'] 
287
288                T = hff_X.shape[0]
289                K = W_list[i][h].shape[1]
290                mf_X_hff = np.hstack((np.ones((T, 1)), np.full((T, mfmodel.M), np.nan)))
291
292                # iterate over dates
293                hfforecast = pd.DataFrame(data=np.full((T, K), np.nan), index=hfforecast_dates)
294                for j, nd in enumerate(hfforecast_dates):
295                    mf_X_hff[j,1:] = hff_X[j,:]
296                    
297                    W_j = W_list[i][((j + h) % fit['freqratio'])]
298                    hfforecast.loc[nd] = np.hstack([1, hff_X[j,:]]) @ W_j
299
300            else:
301                T = hff_X.shape[0]
302                mf_X_hff = np.hstack((np.ones((T, 1)), hff_X))
303                hfforecast = pd.DataFrame(data=(mf_X_hff @ W), index=hfforecast_dates)
304
305            # output
306            hffor.append({
307                's': s,
308                'forecast': hfforecast
309            })
310    
311
312    return({
313        'model': "MFESN",
314        'method': "mfNowcast",
315        'states': Xfhff,
316        'mf_X_n': mf_X_hff,
317        'forecast': hffor,
318        # additional info
319        'hff_X_dates': hff_X_dates, 
320        'init': init,
321    })

419            fstates_j = pd.DataFrame(
420                data=np.vstack([init_list[j], fstates_j]), 
421                index=fstates_dates_j,
422            )
