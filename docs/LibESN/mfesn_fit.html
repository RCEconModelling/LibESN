 33    def __init__(self, Lambda) -> None:
 34        # parameter check
 35        if type(Lambda) in [tuple, list]:
 36            L = []
 37            for i, L0 in enumerate(Lambda):
 38                if type(L0) in [tuple, list]:
 39                    l = []
 40                    for j, l0 in enumerate(L0):
 41                        try:
 42                            l.append(ridge_penalty_check(l0))
 43                        except:
 44                            raise ValueError(f"Lambda failed check at index [{i}][{j}]")
 45                    L.append(l)
 46                else:
 47                    try:
 48                        L1 = ridge_penalty_check(L0)
 49                    except:
 50                        raise ValueError(f"Lambda failed check at index {i}")
 51                    L.append(L1)
 52        else:
 53            try:
 54                L = ridge_penalty_check(Lambda)
 55            except:
 56                raise ValueError("Lambda failed check")
 57            
 58        self.Lambda = L

 60    def fit(
 61        self, 
 62        mfmodel: MFESN, 
 63        train_data: Union[tuple, list], 
 64        full=True, 
 65        **kwargs
 66    ) -> dict:
 67        # unwrap
 68        inputs, targets = train_data
 69
 70        # prepare data
 71        Z, Z_dates = pd_data_prep(inputs)
 72        Y, Y_dates = pd_data_prep(targets)
 73
 74        # handle kwargs
 75        init = None if not 'init' in kwargs.keys() else kwargs['init']
 76        burnin = [0 for _ in range(mfmodel.size)] if not 'burnin' in kwargs.keys() else kwargs['burnin']
 77        for bn in burnin:
 78            assert bn >= 0, "burnin must be a non-negative integer"
 79        
 80        Lambda_models_ar = np.zeros((mfmodel.size)) if not 'Lambda_models_ar' in kwargs.keys() else kwargs['Lambda_models_ar']
 81        
 82        # states
 83        mf_model_states = mf_dt_states(
 84            input=Z, 
 85            mfmodel=mfmodel, 
 86            dates=Z_dates, 
 87            init=init, 
 88            burnin=burnin,
 89        )
 90
 91        # slice and stack model states plus slicing dates
 92        mf_X0, mf_X0_dates = mf_dt_states_to_matrix(
 93            states=mf_model_states, 
 94            ref_dates=Y_dates, 
 95            states_join=mfmodel.states_join, 
 96            states_lags=mfmodel.states_lags,
 97        )
 98
 99        # fit target
100        W = mfridge(
101            X=mf_X0[0:-1,:], 
102            Y=Y[1:,:],
103            Lambda=self.Lambda,
104            N=mfmodel.N,
105        )
106
107        # optional
108        # fit individual autoregressive models for mfmodel components 
109        # (needed e.g. for multi-step forecasting)
110        if full:
111            W_models_ar = []
112            for j, Xj in enumerate(mf_model_states):
113                # reduce to ndarray
114                Xj_ = Xj.iloc[0:-1,:].to_numpy()
115                Yj_ = Z[j][1:,:]
116                # fit
117                Wj_ = ridge(X=Xj_, Y=Yj_, Lambda=Lambda_models_ar[j])
118                W_models_ar.append(Wj_)
119        else:
120            W_models_ar = None
121
122        # outputs
123        T = mf_X0.shape[0]-1
124        mf_X = np.hstack((np.ones((T, 1)), mf_X0[0:-1,:]))
125        mf_Y = pd.DataFrame(data=Y[1:,:], index=Y_dates[1:])
126        mf_Y_fit = pd.DataFrame(data=(mf_X @ W), index=Y_dates[1:])
127        residuals = mf_Y - mf_Y_fit
128        MSE = np.mean(np.square(residuals.to_numpy()))
129
130        return({
131            'model': "MFESN",
132            'method': "mfRidgeFit.fit",
133            'W': W,
134            'states': mf_model_states,
135            'mf_X': mf_X0,
136            'mf_X_dates': mf_X0_dates,
137            'mf_Y': mf_Y,
138            'mf_Y_fit': mf_Y_fit,
139            'residuals': residuals,
140            'MSE': MSE,
141            # method info
142            'Lambda': self.Lambda,
143            # additional info
144            'full': full,
145            'W_models_ar': W_models_ar,
146            'Lambda_models_ar': Lambda_models_ar,
147            'init': init,
148            'burnin': burnin,
149        })

273                's': s,
274                'W': W_s,
275                'mf_X': mf_X,
276                'mf_Y': mf_Y,
277                'mf_Y_fit': mf_Y_fit,
278                'residuals': residuals,
279                'MSE': MSE,
280            })

391    def fitDirectHighFreq(
392        self, 
393        mfmodel: MFESN, 
394        train_data: Union[tuple, list], 
395        freqratio, 
396        steps=0, 
397        **kwargs
398    ) -> dict:
399        """
400        Fit MFESN using direct regressions at fixed high-frequency steps over
401        each low-frequency interval. This is explicitly designed for nowcasting
402        as the targets used are always the closest LF future observations (w.r.t
403        current HF time).
404        mfmodel: MFESN model.
405        train_data: fitting dataset.
406        freqratio: fixed integer ratio of HF to LF observations. 
407        steps: number of fit steps. Default is s=0 because unline in the low-freq.
408               setup, HF regressors can be observed at distance of less than 1
409               LF interval from the target c.f. nowcasting.
410        """
411        # unwrap
412        inputs, targets = train_data
413
414        # prepare data
415        Z, Z_dates = pd_data_prep(inputs)
416        Y, Y_dates = pd_data_prep(targets)
417
418        # handle freqratio
419        assert freqratio >= 1, "freqratio must be a positive integer"
420
421        # handle kwargs
422        init = None if not 'init' in kwargs.keys() else kwargs['init']
423        burnin = [0 for _ in range(mfmodel.size)] if not 'burnin' in kwargs.keys() else kwargs['burnin']
424        for bn in burnin:
425            assert bn >= 0, "burnin must be a non-negative integer"
426
427        # handle steps
428        if type(steps) is int:
429            # NOTE: here we start the range at 0, see def of steps above
430            steps = range(0, steps+1)
431        elif type(steps) in [tuple, list]:
432            for s in steps:
433                assert type(s) is int, "steps must be a tuple/list of integers"
434        else:
435            raise ValueError("steps must be an integer or a tuple/list of integers")
436
437        # states
438        mf_model_states = mf_dt_states(
439            input=Z, 
440            mfmodel=mfmodel, 
441            dates=Z_dates, 
442            init=init, 
443            burnin=burnin,
444        )
445
446        # find high-frequency date slices
447        max_freq_dates = pd_max_freq_dates(Z_dates)
448        max_freq_dslices = []
449        for k in range(freqratio):
450            dslice_k = []
451            l = None
452            if k == 0:
453                # start with closest future dates to Y_dates
454                for d in Y_dates[:-1]:
455                    cfd, l = closest_future_date(max_freq_dates, d, cutoff=l, strict=True)
456                    dslice_k.append(cfd)
457                    # adjust cutoff
458                    l += 2*freqratio
459            else:
460                # iterate, closest future date to previous slice
461                for d in dslice_0:
462                    cfd, l = closest_future_date(max_freq_dates, d, cutoff=l, strict=True)
463                    dslice_k.append(cfd)
464                    # adjust cutoff
465                    l += 2*freqratio
466            # append
467            dslice_0 = dslice_k
468            max_freq_dslices.append(pd.to_datetime(dslice_k))
469
470        # fit model
471        fits = []
472        for i, s in enumerate(steps):
473            # step lambda
474            # NOTE: must be able to hande nested lists e.g. [[1,2],]
475            #       where the outer level list enumerates over steps,
476            #       and the inner level list enumerates over freq. ratio
477            Lambda_s = None
478            if type(self.Lambda) in [tuple, list]:
479                if len(self.Lambda) == len(steps):
480                    Lambda_s = self.Lambda[i]
481                elif len(self.Lambda) == 1 and type(self.Lambda[0]) in [tuple, list]:
482                    Lambda_s = self.Lambda[0]
483            else:
484                Lambda_s = self.Lambda
485
486            freq_fits_s = []
487            for k in range(freqratio):
488                # sub-step lambda
489                Lambda_k = Lambda_s[k] if type(Lambda_s) in [tuple, list] else Lambda_s
490
491                # slice and stack model states plus slicing dates
492                mf_X0_k, mf_X0_dates_k = mf_dt_states_to_matrix(
493                    states=mf_model_states, 
494                    ref_dates=max_freq_dslices[k], 
495                    states_join=mfmodel.states_join, 
496                    states_lags=mfmodel.states_lags,
497                )
498
499                # print(f"+ {s} - {k}")
500
501                # fit target
502                # NOTE: for regressors, need to index with 0:(-s)
503                #       because the 1st step in HF means that we are
504                #       targeting the next available LF observation,
505                #       which is less than 1 LF interval away c.f. nowcasting
506                mf_X0_k_s = mf_X0_k if s == 0 else mf_X0_k[0:(-s),:]
507                Y_k_s = Y[(1+s):,:]
508
509                W_k = mfridge(
510                    X=mf_X0_k_s, 
511                    Y=Y_k_s,
512                    Lambda=Lambda_k,
513                    N=mfmodel.N,
514                )
515
516                # outputs
517                T = mf_X0_k_s.shape[0]
518                mf_X_k = np.hstack((np.ones((T, 1)), mf_X0_k_s))
519                mf_Y_k = pd.DataFrame(data=Y[(1+s):,:], index=Y_dates[(1+s):])
520                mf_Y_fit = pd.DataFrame(data=(mf_X_k @ W_k), index=Y_dates[(1+s):])
521                residuals = mf_Y_k - mf_Y_fit
522                MSE = np.mean(np.square(residuals.to_numpy()))
523
524                # outputs
525                freq_fits_s.append({
526                    'k': k,
527                    'hf_lag': (freqratio-k-1),
528                    'W': W_k,
529                    'mf_X': mf_X_k,
530                    'mf_X_dates': mf_X0_dates_k,
531                    'mf_Y': mf_Y_k,
532                    'mf_Y_fit': mf_Y_fit,
533                    'residuals': residuals,
534                    'MSE': MSE,
535                    # additional info
536                    'Lambda': Lambda_k,
537                })
538
539            # steps output
540            fits.append({
541                's': s,
542                'freq_fits': freq_fits_s
543            })
544
545        return({
546            'model': "MFESN",
547            'method': "mfRidgeFit.fitDirectHighFreq",
548            'fits': fits,
549            'states': mf_model_states,
550            # method info
551            'Lambda': self.Lambda,
552            # additional info
553            'steps': steps,
554            'freqratio': freqratio,
555            'max_freq_slices': max_freq_dslices,
556            'init': init,
557            'burnin': burnin,
558        })

566        self,
567        mfmodel: MFESN, 
568        train_data: Union[tuple, list], 
569        step=1,
570        **kwargs
571    ) -> dict:
572        # unwrap
573        inputs, targets = train_data
574
575        # prepare data
576        Z, Z_dates = pd_data_prep(inputs)
577        Y, Y_dates = pd_data_prep(targets)
578
579        # handle kwargs
580        init = None if not 'init' in kwargs.keys() else kwargs['init']
581        burnin = [0 for _ in range(mfmodel.size)] if not 'burnin' in kwargs.keys() else kwargs['burnin']
582        for bn in burnin:
583            assert bn >= 0, "burnin must be a non-negative integer"
584        
585        test_size = 1 if not 'test_size' in kwargs.keys() else kwargs['test_size']
586        min_train_size = 1 if not 'min_train_size' in kwargs.keys() else kwargs['min_train_size']
587        max_train_size = None if not 'max_train_size' in kwargs.keys() else kwargs['max_train_size']
588        overlap = False if not 'overlap' in kwargs.keys() else kwargs['overlap']
589        
590        optim = 'scipy' if not 'optim' in kwargs.keys() else kwargs['optim']
591        
592        verbose = False if not 'verbose' in kwargs.keys() else kwargs['verbose']
593
594        # handle steps
595        if not type(step) is int:
596            raise ValueError("steps must be an integer or a tuple, list or range of integers")
597        assert step >= 0, "step must be a non-negative integer"
598        
599        # states
600        mf_model_states = mf_dt_states(
601            input=Z, 
602            mfmodel=mfmodel, 
603            dates=Z_dates, 
604            init=init, 
605            burnin=burnin,
606        )
607
608        # slice and stack model states plus slicing dates
609        mf_X0, _ = mf_dt_states_to_matrix(
610            states=mf_model_states, 
611            ref_dates=Y_dates, 
612            states_join=mfmodel.states_join, 
613            states_lags=mfmodel.states_lags,
614        )
615
616        # slice step
617        mf_X0_cv = mf_X0[0:(-step),:]
618        mf_Y_cv = Y[step:,:]
619
620        # cv splits
621        tscv = ShiftTimeSeriesSplit(
622            length=mf_X0_cv.shape[0],
623            test_size=test_size,
624            min_train_size=min_train_size,
625            max_train_size=max_train_size,
626            overlap=overlap,
627        )
628        iter_tscv = iter(tscv)
629
630        # objective function
631        Nmods = mfmodel.N
632
633        def cv_obj(cv_lambda):
634            # Lambda = np.exp(cv_lambda)
635            Lambda = cv_lambda
636
637            cv_RSS = 0
638            for train_idx, test_idx in iter_tscv:
639                W = jit_kron_mfridge(
640                    X=mf_X0_cv[train_idx,:], 
641                    Y=mf_Y_cv[train_idx,:], 
642                    Lambda=Lambda,
643                    N=Nmods,
644                )
645                residuals = mf_Y_cv[test_idx] - np.hstack([np.ones((len(test_idx), 1)), mf_X0_cv[test_idx,:]]) @ W
646                cv_RSS += np.sum(np.square(residuals))
647

898def mfridge(X: np.ndarray, Y: np.ndarray, Lambda=None, N=None):
899    """
900    Ridge regression for a multi-frequency setup i.e. if N is a 
901    list of dimensions, and Lambda and N have the same length, then
902    the final penalty is given by diag(kron(Lambda, rep(1, N)))
903    """
904    # sanity checks
905    Tx, Kx = X.shape
906    Ty, _  = Y.shape
907    assert Tx == Ty, "Shapes of X and Y non compatible"
908
909    if Lambda is None:
910        print("[-!-] mfridge() called with Lambda set to None!")
911        L = np.array(0)
912    else:
913        L = np.array(Lambda).squeeze()
914
915    if L.shape == ():
916        # Lambda is a scalar 
917        if L < 1e-12:
918            V = np.hstack((np.ones((Tx, 1)), X))
919            res = np.linalg.lstsq(V, Y, rcond=None)
920            W = res[0]
921        else:
922            W = np.linalg.solve(((X.T @ X / Tx) + L * np.eye(Kx)), (X.T @ Y / Ty))
923            a = np.mean(Y.T, axis=1) - W.T @ np.mean(X.T, axis=1).T
924            W = np.vstack((a, W))
925    elif not N is None:
926        N = np.array(N).astype(int)
927        assert len(Lambda) == len(N), "Lambda and N should have the same length"
928        assert np.min(N) > 0, "N should contain only positive integers"
929        assert np.sum(N) == Kx, "Elements of N should sum up to the state dimension of X"
930        l = np.concatenate([L[i] * np.ones(N[i]) for i in range(len(N))])
931        W = np.linalg.solve(((X.T @ X / Tx) + np.diag(l)), (X.T @ Y / Ty))
932        a = np.mean(Y.T, axis=1) - W.T @ np.mean(X.T, axis=1).T
933        W = np.vstack((a, W))
934    else:
935        raise ValueError("Could not run ridge regression")
936
937    return W

940def jit_kron_mfridge(X: np.ndarray, Y: np.ndarray, Lambda, N):
941    Tx = X.shape[0]
942    Ty = Y.shape[0]
943    L = np.array(Lambda)
944    N = np.array(N).astype(int)
945    l = np.concatenate([L[i] * np.ones(N[i]) for i in range(len(N))])
946    W = np.linalg.solve(((X.T @ X / Tx) + np.diag(l)), (X.T @ Y / Ty))
947    a = np.mean(Y.T, axis=1) - W.T @ np.mean(X.T, axis=1).T
948    W = np.vstack((a, W))
949    
950    return W
