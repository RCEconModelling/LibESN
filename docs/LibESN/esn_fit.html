338# non-negativity check function
339def ridge_penalty_check(L) -> np.ndarray:
340    """Check if the ridge penalty -- scalar, vector or matrix -- is nonnegative."""
341    L = np.squeeze(np.copy(L))
342
343    if L.shape == ():
344        assert L >= 0, "L must be a nonnegative scalar"
345    elif len(L.shape) == 1:
346        assert np.min(L) >= 0, "L must be a nonnegative vector"
347    else:
348        assert np.min(npLA.eigvals(L)) >= 0, "L must be nonnegative definite"
349
350    return L

352def ridge(X: np.ndarray, Y: np.ndarray, Lambda=None):
353    r"""
354    Ridge regression. Will yield a matrix `W` of estimated regression coefficients.
355    Note that this model adapts to the shape of the input Lambda.
356    
357    Assume that `X` is a $T \times N$ state matrix, where $T$ is the sample size, and 
358    `Y` is a $T \times M$ matrix of targets/outputs, for $M \geq 1$. 
359    The following cases are supported:
360
361    + If `Lambda` is a scalar,
362        $$ \tilde{W} = \left(\frac{X' X}{T} + \textnormal{Lambda} * I \right)^{-1} \frac{X' Y}{T} $$
363    + If `Lambda` is a vector of length $N$,
364        $$ \tilde{W} = \left(\frac{X' X}{T} + \textnormal{diag}(\textnormal{Lambda}) \right)^{-1} \frac{X' Y}{T} $$
365    + If `Lambda` is a $N \times N$ matrix,
366        $$ \tilde{W} = \left(\frac{X' X}{T} + \textnormal{Lambda} \right)^{-1} \frac{X' Y}{T} $$
367
368    Finally, the intercept term is estimated as:
369    $$ a = \bar{Y} - W' \bar{X} $$
370    where $\bar{Y}$ and $\bar{X}$ are the column-wise means of `Y` and `X` (i.e. sample averages), respectively.
371
372    The output `W` is a $(N+1) \times M$ matrix given by $W := (a, \tilde{W})$.
373    """
374    # sanity checks
375    Tx, Kx = X.shape
376    Ty, _  = Y.shape
377    assert Tx == Ty, "Shapes of X and Y non compatible"
378
379    L = np.array(0) if Lambda is None else Lambda
380
381    # Regression matrices
382    if L.shape == ():
383        # Lambda is a scalar 
384        if L < 1e-12:
385            V = np.hstack((np.ones((Tx, 1)), X))
386            res = np.linalg.lstsq(V, Y, rcond=None)
387            W = res[0]
388        else:
389            W = np.linalg.solve(((X.T @ X / Tx) + L * np.eye(Kx)), (X.T @ Y / Ty))
390            a = np.mean(Y.T, axis=1) - W.T @ np.mean(X.T, axis=1).T
391            W = np.vstack((a, W))
392    else:
393        if len(L.shape) == 1:
394            assert len(L) == Kx, f"Lambda is not scalar, vector or 2D matrix with Gram shape ({Kx},{Kx}), found shape {Lambda.shape}"
395            L = np.diag(L)
396        else:
397            assert L.shape == (Kx, Kx), f"Lambda is not scalar, vector or 2D matrix with Gram shape ({Kx},{Kx}), found shape {Lambda.shape}"
398        W = np.linalg.solve(((X.T @ X / Tx) + L), (X.T @ Y / Ty))
399        a = np.mean(Y.T, axis=1) - W.T @ np.mean(X.T, axis=1).T
400        W = np.vstack((a, W))
401
402    return W

405def jit_ridge(X: np.ndarray, Y: np.ndarray, L: np.ndarray):
406    """
407    JIT-compiled ridge regression. Will yield a matrix `W` of estimated regression coefficients.
408    
409    Note that this function only supports the case where `Lambda` is a scalar.
410    """
411    T, K = X.shape
412    W = np.linalg.solve(((X.T @ X / T) + L * np.eye(K)), (X.T @ Y / T))
413    a = np.mean(Y.T, axis=1) - W.T @ np.mean(X.T, axis=1).T
414    W = np.vstack((a, W))
415
416    return W

434        self,
435        model: ESN,

439    ) -> dict:
440        r"""
441        Cross-validation to estimate the ridge penalty for **single multistep** ridge fit. 
442        The CV folds are contructed using `libesn.validation.ShiftTimeSeriesSplit`.
443
444        + `model`: ESN model.
445        + `train_data`: tuple or list of training data, where the first element is the input data 
446            and the second element is the target data.
447        + `step`: integer, step (prediction horizon) for the ridge fit. Default is 1.
448
449        Optional keyword arguments:
450        + `test_size`: float, size of the test split. Default is 1.
451        + `min_train_size`: integer, *minimum* size of the training split. Default is 1.
452        + `max_train_size`: integer, *maximum* size of the training split. Default is None.
453        + `overlap`: boolean, whether to allow overlap between training and test splits
454            when the cross-validation folds are built. Default is `False`.
455
456        See `libesn.esn_fit.ridgeFit` for additional keyword arguments `init` and `burnin`.
457        """
458
459        # unwrap
460        inputs, targets = train_data
461
462        # prepare data
463        Z, Z_dates = pd_data_prep(inputs)
464        Y, Y_dates = pd_data_prep(targets)
465
466        # handle kwargs
467        init = None if not 'init' in kwargs.keys() else kwargs['init']
468        burnin = 0 if not 'burnin' in kwargs.keys() else kwargs['burnin']
469        assert burnin >= 0, "burnin must be a non-negative integer"
470
471        test_size = 1 if not 'test_size' in kwargs.keys() else kwargs['test_size']
472        min_train_size = 1 if not 'min_train_size' in kwargs.keys() else kwargs['min_train_size']
473        max_train_size = None if not 'max_train_size' in kwargs.keys() else kwargs['max_train_size']
474        overlap = False if not 'overlap' in kwargs.keys() else kwargs['overlap']
475
476        # handle steps
477        if not type(step) is int:
478            raise ValueError("steps must be an integer or a tuple, list or range of integers")
479        assert step >= 0, "step must be a non-negative integer"
480
481        # collect states
482        model_states = states(
483            Z, 
484            map=model.pars.smap, 
485            A=model.pars.A, C=model.pars.C, zeta=model.pars.zeta, 
486            rho=model.pars.rho, gamma=model.pars.gamma, leak=model.pars.leak, 
487            init=init
488        )
489
490        # slice burn-in periods
491        X0 = model_states[burnin:,]
492        Y = Y[burnin:,]
493
494        # slice step
495        X0_cv = X0[0:(-step),:]
496        Y_cv = Y[step:,:]
497
498        # cv splits
499        tscv = ShiftTimeSeriesSplit(
500            length=X0_cv.shape[0],
501            test_size=test_size,
502            min_train_size=min_train_size,
503            max_train_size=max_train_size,
504            overlap=overlap,
505        )
506        iter_tscv = iter(tscv)
507
508        # objective function
509        def cv_obj(cv_lambda):
510            # Lambda = np.exp(cv_lambda)
511            Lambda = cv_lambda
512
513            cv_RSS = 0
514            for train_idx, test_idx in iter_tscv:
515                W = jit_ridge(X=X0_cv[train_idx,:], Y=Y_cv[train_idx,:], L=Lambda)
516                residuals = Y_cv[test_idx] - np.hstack([np.ones((len(test_idx), 1)), X0_cv[test_idx,:]]) @ W
517                cv_RSS += np.sum(np.square(residuals))
518
519            return cv_RSS
520
521        # optimization
522        problem = FunctionalProblem(
523            1,
524            cv_obj,
525            x0=np.array([1e-2]),
526            xl=np.array([1e-9]),
527            xu=np.array([1e5])
528        )
529
530        result = minimize(
531            problem, 
532            PatternSearch(n_sample_points=50), 
533            #get_termination("n_eval", 500),
534            #get_termination("time", "00:01:00"),
535            #get_termination("ftol", 1e-8),
536            verbose=False, 
537            seed=1203477
538        )
539
540        # output
541        cvLambda = result.X[0]
542
543        return({
544            'model': "ESN",
545            'method': "ridgeFit.cv",
546            'cvLambda': cvLambda,
547            'result': result,
548            # additional info
549            'init': init,
550            'burnin': burnin,
551        })

def ridge(X: numpy.ndarray, Y: numpy.ndarray, Lambda=None): 
353def ridge(X: np.ndarray, Y: np.ndarray, Lambda=None):
354    r"""
355    Ridge regression. Will yield a matrix `W` of estimated regression coefficients.
356    Note that this model adapts to the shape of the input Lambda.
357    
358    Assume that `X` is a $T \times N$ state matrix, where $T$ is the sample size, and 
359    `Y` is a $T \times M$ matrix of targets/outputs, for $M \geq 1$. 
360    The following cases are supported:
361
362    + If `Lambda` is a scalar,
363        $$ \tilde{W} = \left(\frac{X' X}{T} + \textnormal{Lambda} * I \right)^{-1} \frac{X' Y}{T} $$
364    + If `Lambda` is a vector of length $N$,
365        $$ \tilde{W} = \left(\frac{X' X}{T} + \textnormal{diag}(\textnormal{Lambda}) \right)^{-1} \frac{X' Y}{T} $$
366    + If `Lambda` is a $N \times N$ matrix,
367        $$ \tilde{W} = \left(\frac{X' X}{T} + \textnormal{Lambda} \right)^{-1} \frac{X' Y}{T} $$
368
369    Finally, the intercept term is estimated as:
370    $$ a = \bar{Y} - W' \bar{X} $$
371    where $\bar{Y}$ and $\bar{X}$ are the column-wise means of `Y` and `X` (i.e. sample averages), respectively.
372
373    The output `W` is a $(N+1) \times M$ matrix given by $W := (a, \tilde{W})$.
374    """
375    # sanity checks
376    Tx, Kx = X.shape
377    Ty, _  = Y.shape
378    assert Tx == Ty, "Shapes of X and Y non compatible"
379
380    L = np.array(0) if Lambda is None else Lambda
381
382    # Regression matrices
383    if L.shape == ():
384        # Lambda is a scalar 
385        if L < 1e-12:
386            V = np.hstack((np.ones((Tx, 1)), X))
387            res = np.linalg.lstsq(V, Y, rcond=None)
388            W = res[0]
389        else:
390            W = np.linalg.solve(((X.T @ X / Tx) + L * np.eye(Kx)), (X.T @ Y / Ty))
391            a = np.mean(Y.T, axis=1) - W.T @ np.mean(X.T, axis=1).T
392            W = np.vstack((a, W))
393    else:
394        if len(L.shape) == 1:
395            assert len(L) == Kx, f"Lambda is not scalar, vector or 2D matrix with Gram shape ({Kx},{Kx}), found shape {Lambda.shape}"
396            L = np.diag(L)
397        else:
398            assert L.shape == (Kx, Kx), f"Lambda is not scalar, vector or 2D matrix with Gram shape ({Kx},{Kx}), found shape {Lambda.shape}"
399        W = np.linalg.solve(((X.T @ X / Tx) + L), (X.T @ Y / Ty))
400        a = np.mean(Y.T, axis=1) - W.T @ np.mean(X.T, axis=1).T
401        W = np.vstack((a, W))
402
403    return W
