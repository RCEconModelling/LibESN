 25# non-negativity check function
 26def ridge_penalty_check(L) -> np.ndarray:
 27    L = np.squeeze(np.copy(L))
 28
 29    if L.shape == ():
 30        assert L >= 0, "L must be a nonnegative scalar"
 31    elif len(L.shape) == 1:
 32        assert np.min(L) >= 0, "L must be a nonnegative vector"
 33    else:
 34        assert np.min(npLA.eigvals(L)) >= 0, "L must be nonnegative definite"
 35
 36    return L

341def ridge(X: np.ndarray, Y: np.ndarray, Lambda=None):
342    # sanity checks
343    Tx, Kx = X.shape
344    Ty, _  = Y.shape
345    assert Tx == Ty, "Shapes of X and Y non compatible"
346
347    L = np.array(0) if Lambda is None else Lambda
348
349    # Regression matrices
350    if L.shape == ():
351        # Lambda is a scalar 
352        if L < 1e-12:
353            V = np.hstack((np.ones((Tx, 1)), X))
354            res = np.linalg.lstsq(V, Y, rcond=None)
355            W = res[0]
356        else:
357            W = np.linalg.solve(((X.T @ X / Tx) + L * np.eye(Kx)), (X.T @ Y / Ty))
358            a = np.mean(Y.T, axis=1) - W.T @ np.mean(X.T, axis=1).T
359            W = np.vstack((a, W))
360    else:
361        if len(L.shape) == 1:
362            assert len(L) == Kx, f"Lambda is not scalar, vector or 2D matrix with Gram shape ({Kx},{Kx}), found shape {Lambda.shape}"
363            L = np.diag(L)
364        else:
365            assert L.shape == (Kx, Kx), f"Lambda is not scalar, vector or 2D matrix with Gram shape ({Kx},{Kx}), found shape {Lambda.shape}"
366        W = np.linalg.solve(((X.T @ X / Tx) + L), (X.T @ Y / Ty))
367        a = np.mean(Y.T, axis=1) - W.T @ np.mean(X.T, axis=1).T
368        W = np.vstack((a, W))
369
370    return W

373def jit_ridge(X: np.ndarray, Y: np.ndarray, L: np.ndarray):
374    T, K = X.shape
375    W = np.linalg.solve(((X.T @ X / T) + L * np.eye(K)), (X.T @ Y / T))
376    a = np.mean(Y.T, axis=1) - W.T @ np.mean(X.T, axis=1).T
377    W = np.vstack((a, W))
378
379    return W

344    Tx, Kx = X.shape
345    Ty, _  = Y.shape
346    assert Tx == Ty, "Shapes of X and Y non compatible"
347
348    L = np.array(0) if Lambda is None else Lambda
349
350    # Regression matrices
351    if L.shape == ():
352        # Lambda is a scalar 
353        if L < 1e-12:
354            V = np.hstack((np.ones((Tx, 1)), X))
355            res = np.linalg.lstsq(V, Y, rcond=None)
356            W = res[0]
357        else:
358            W = np.linalg.solve(((X.T @ X / Tx) + L * np.eye(Kx)), (X.T @ Y / Ty))
359            a = np.mean(Y.T, axis=1) - W.T @ np.mean(X.T, axis=1).T
360            W = np.vstack((a, W))
361    else:
362        if len(L.shape) == 1:
363            assert len(L) == Kx, f"Lambda is not scalar, vector or 2D matrix with Gram shape ({Kx},{Kx}), found shape {Lambda.shape}"
364            L = np.diag(L)
365        else:
366            assert L.shape == (Kx, Kx), f"Lambda is not scalar, vector or 2D matrix with Gram shape ({Kx},{Kx}), found shape {Lambda.shape}"
367        W = np.linalg.solve(((X.T @ X / Tx) + L), (X.T @ Y / Ty))
368        a = np.mean(Y.T, axis=1) - W.T @ np.mean(X.T, axis=1).T
369        W = np.vstack((a, W))
370
